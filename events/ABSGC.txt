country_event = {
	id = ABSGC.3
	title = ABSGC.3.t
	#always = yes
	hidden = yes
	fire_only_once = no
	desc = ABSGC.3.d


	trigger = {
		any_country = {
		has_idea = war_economy
		NOT = {
			has_idea = tot_economic_mobilisation
		}
        has_war = yes
		NOT = {
			has_country_flag = One_time_tortal_war_flag
		}
		}

	}
	
	picture = GFX_report_event_ace_promoted
	
	immediate = {
		random_country = {
			limit = {
				has_war = yes
				has_idea = war_economy
				NOT = {
					has_idea = tot_economic_mobilisation
				}
			}
			add_ideas = tot_economic_mobilisation
			set_country_flag = One_time_tortal_war_flag
		}
	}
}
